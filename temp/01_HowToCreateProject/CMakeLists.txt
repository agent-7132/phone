###############################################################################
# ESP-IDF 项目主构建配置文件
# 
# 该文件位于项目根目录，是 ESP-IDF 构建系统的入口文件。它负责：
# 1. 指定 CMake 最低版本要求
# 2. 引入 ESP-IDF 构建系统核心模块
# 3. 定义项目名称
# 
# ESP-IDF 构建系统基于 CMake，采用组件化架构。项目的构建流程如下：
# 1. CMake 读取此文件，执行 cmake_minimum_required 和 include 指令
# 2. project.cmake 加载 ESP-IDF 的构建系统配置
# 3. 扫描项目目录，查找所有组件的 CMakeLists.txt
# 4. 根据组件依赖关系生成构建文件
# 5. 执行编译、链接生成最终固件
# 
# 官方文档：https://docs.espressif.com/projects/esp-idf/en/latest/api-guides/build-system.html
###############################################################################

#-----------------------------------------------------------------------------
# 指定 CMake 最低版本要求
# 
# ESP-IDF 5.x 版本要求 CMake 3.16 或更高版本。
# 该指令必须放在文件最开头，在任何其他 CMake 指令之前。
# 
# 如果系统安装的 CMake 版本低于此要求，构建将失败并提示错误信息。
#-----------------------------------------------------------------------------
cmake_minimum_required(VERSION 3.16)

#-----------------------------------------------------------------------------
# 引入 ESP-IDF 构建系统核心模块
# 
# 该指令加载 ESP-IDF 提供的 project.cmake 文件，该文件包含了：
# - 组件扫描和依赖解析逻辑
# - 编译选项和链接器配置
# - 目标平台（ESP32/P4/S3 等）支持
# - 固件生成和烧录工具链
# 
# $ENV{IDF_PATH} 环境变量指向 ESP-IDF SDK 的安装路径，必须正确设置。
# 
# 注意：该指令必须在 project() 指令之前调用。
#-----------------------------------------------------------------------------
include($ENV{IDF_PATH}/tools/cmake/project.cmake)

#-----------------------------------------------------------------------------
# 定义项目名称
# 
# project() 指令定义项目名称为 "HowToCreateProject"。
# 该名称将用于：
# - 生成的固件文件名（如 HowToCreateProject.elf、HowToCreateProject.bin）
# - 构建目录名称
# - IDE 中的项目标识
# 
# ESP-IDF 会根据项目名称自动生成以下目标：
# - build: 编译项目生成固件
# - flash: 将固件烧录到设备
# - monitor: 启动串口监视器查看日志
# - clean: 清理构建目录
# 
# 项目名称应遵循以下规范：
# - 仅包含字母、数字和下划线
# - 不包含空格和特殊字符
# - 不以下划线开头
#-----------------------------------------------------------------------------
project(HowToCreateProject)
